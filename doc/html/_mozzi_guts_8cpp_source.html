00028 // Interrupts:
00029 // Timer2 has highest priority (on a 328), but you can't have a flexible interrupt rate as well as pwm.
00030 // The TimerOne library is more flexible and allows both pwm and interrupts,
00031 // by using ICR1 for interrupt rate and OCR1A (pin 9) (or OCR1B pin 10) for pwm level
00032 // So we use Timer1 for audio.
00033 // Ideally we'd use Timer0 for control because it has lowest priority, but Arduino uses
00034 // it for time and delay functions, so we use Timer 2 for control and hope it won't
00035 // interfere with the audio interrupt.
00036 // TODO: Timer0 is messed up anyway, it seems, so maybe it's better to
00037 // use it for control and leave Timer2 available for other things?

00086 void audioHook()
00087 {
00088 
00089                 static unsigned char in_pos = 0;
00090                 static unsigned int num_in = 0;
00091                 unsigned int gap;
00092                 // Upon entering the block the Global Interrupt Status flag in SREG is disabled, and re-enabled upon exiting the block
00093                 ATOMIC_BLOCK(ATOMIC_RESTORESTATE)
00094                 {
00095                                 // Atomic access to 16 bit num_out, which is changed in outputAudio(), called in the audio interrupt.
00096                                 // It's atomic because we don't want num_out to change half-way through reading it.
00097                                 gap = num_in - num_out; // wraps to a big number if it's negative
00098                 }
00099                 if (gap < BUFFER_NUM_CELLS)
00100                 {
00101                                 in_pos++;
00102                                 num_in++;
00103                                 output_buffer[in_pos] = updateAudio() + AUDIO_BIAS;
00104                 }
00105 
00106 }
