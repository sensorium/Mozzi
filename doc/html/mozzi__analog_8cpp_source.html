00145 int adcGetResult()
00146 {
00147                 //unsigned char low, high;
00148                 int out;
00149 #if defined(ADCSRA) && defined(ADCL)
00150                 // ADSC is cleared when the conversion finishes
00151                 while (bit_is_set(ADCSRA, ADSC))
00152                                 ;
00153 
00154                 // we have to read ADCL first; doing so locks both ADCL
00155                 // and ADCH until ADCH is read.  reading ADCL second would
00156                 // cause the results of each conversion to be discarded,
00157                 // as ADCL and ADCH would be locked when it completed.
00158                 //low  = ADCL;
00159                 //high = ADCH;
00160                 out = ADC;
00161 #else
00162                 // we dont have an ADC, return 0
00163                 //low  = 0;
00164                 //high = 0;
00165                 out = 0;
00166 #endif
00167 
00168                 // combine the two bytes
00169                 //return (high << 8) | low;
