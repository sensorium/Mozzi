00078 Q16n16  Q16n16_mtof(Q16n16 midival_fractional)
00079 {
00080                 Q16n16 diff_fraction;
00081                 unsigned char index = midival_fractional >> 16;
00082                 unsigned int fraction = (unsigned int) midival_fractional; // keeps low word
00083                 Q16n16 freq1 = (Q16n16) pgm_read_dword(midiToFreq + index);
00084                 Q16n16 freq2 = (Q16n16) pgm_read_dword(midiToFreq + (index+1));
00085                 Q16n16 difference = freq2 - freq1;
00086                 if (difference>=65536)
00087                 {
00088                                 diff_fraction = ((difference>>8) * fraction) >> 8;
00089                 }
00090                 else
00091                 {
00092                                 diff_fraction = (difference * fraction) >> 16;
00093                 }
00094                 return (Q16n16) (freq1+ diff_fraction);

00351 int receiveAnalogRead()
00352 {
00353                 unsigned char low, high;
00354 #if defined(ADCSRA) && defined(ADCL)
00355                 // ADSC is cleared when the conversion finishes
00356                 while (bit_is_set(ADCSRA, ADSC))
00357                                 ;
00358 
00359                 // we have to read ADCL first; doing so locks both ADCL
00360                 // and ADCH until ADCH is read.  reading ADCL second would
00361                 // cause the results of each conversion to be discarded,
00362                 // as ADCL and ADCH would be locked when it completed.
00363                 low  = ADCL;
00364                 high = ADCH;
00365 #else
00366                 // we dont have an ADC, return 0
00367                 low  = 0;
00368                 high = 0;
00369 #endif
00370 
00371                 // combine the two bytes
00372                 return (high << 8) | low;
00373 }
