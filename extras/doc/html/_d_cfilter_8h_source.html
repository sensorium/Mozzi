50  DCfilter(float pole):acc(0),prev_x(0),prev_y(0)

55 /* almost original
56  // timing: 20us
57  int next(int x)
58  {
59  setPin13High();
60  acc -= prev_x;
61  prev_x = (long)x<<15;
62  acc += prev_x;
63  acc -= A*prev_y;
64  prev_y = acc>>15; // quantization happens here
65  int filtered = (int)prev_y;
66  // acc has y[n] in upper 17 bits and -e[n] in lower 15 bits
67  setPin13Low();
68  return filtered;
69  }
70  */
71 
77  // timing :8us
78  inline
79  int next(int x)
80  {
81  acc += ((long)(x-prev_x)<<16)>>1;
82  prev_x = x;
83  acc -= (long)A*prev_y; // acc has y[n] in upper 17 bits and -e[n] in lower 15 bits
84  prev_y = (acc>>16)<<1; // faster than >>15 but loses bit 0
85  if (acc & 32784) prev_y += 1; // adds 1 if it was in the 0 bit position lost in the shifts above
86  return prev_y;
87  }
88 
89 private:
90  long acc;
91  int prev_x, prev_y,A;
92 };
