30 /*
31 ATmega328 technical manual, Section 12.7.4:
32 The dual-slope operation [of phase correct pwm] has lower maximum operation
33 frequency than single slope operation. However, due to the symmetric feature
34 of the dual-slope PWM modes, these modes are preferred for motor control
35 applications.
36 Due to the single-slope operation, the operating frequency of the
37 fast PWM mode can be twice as high as the phase correct PWM mode that use
38 dual-slope operation. This high frequency makes the fast PWM mode well suited
39 for power regulation, rectification, and DAC applications. High frequency allows
40 physically small sized external components (coils, capacitors)..
41 
42 DAC, that's us! Fast PWM.
43 
44 PWM frequency tests
45 62500Hz, single 8 or dual 16 bits, bad aliasing
46 125000Hz dual 14 bits, sweet
47 250000Hz dual 12 bits, gritty, if you're gonna have 2 pins, have 14 bits
48 500000Hz dual 10 bits, grittier
49 16384Hz single nearly 9 bits (original mode) not bad for a single pin, but carrier freq noise can be an issue
50 */

267 static void startAudioHiFi()
268 {
269  backupPreMozziTimer1();
270 
271  // pwm on timer 1
272  pinMode(AUDIO_CHANNEL_1_HIGHBYTE_PIN, OUTPUT); // set pin to output for audio, use 3.9k resistor
273  pinMode(AUDIO_CHANNEL_1_LOWBYTE_PIN, OUTPUT); // set pin to output for audio, use 499k resistor
274 
275  Timer1.initializeCPUCycles(16000000UL/125000, FAST); // set period for 125000 Hz fast pwm carrier frequency = 14 bits
276 
277  Timer1.pwm(AUDIO_CHANNEL_1_HIGHBYTE_PIN, 0); // pwm pin, 0% duty cycle, ie. 0 signal
278  Timer1.pwm(AUDIO_CHANNEL_1_LOWBYTE_PIN, 0); // pwm pin, 0% duty cycle, ie. 0 signal
279 
280  backupMozziTimer1();
281 
282  // audio output interrupt on timer 2, sets the pwm levels of timer 1
283  setupTimer2();
284 }

394 /* Sets up Timer 0 for control interrupts. This is the same for all output
395 options Using Timer0 for control disables Arduino's time functions but also
396 saves on the interrupts and blocking action of those functions. May add a config
397 option for Using Timer2 instead if needed. (MozziTimer2 can be re-introduced for
398 that). */
399 static void startControl(unsigned int control_rate_hz)
400 {
401  // backup pre-mozzi register values
402  pre_mozzi_TCCR0A = TCCR0A;
403  pre_mozzi_TCCR0B = TCCR0B;
404  pre_mozzi_OCR0A = OCR0A;
405  pre_mozzi_TIMSK0 = TIMSK0;
406 
407  TimerZero::init(1000000/control_rate_hz,updateControlWithAutoADC); // set period, attach updateControlWithAutoADC()
408  TimerZero::start();
409 
410  // backup mozzi register values for unpausing later
411  mozzi_TCCR0A = TCCR0A;
412  mozzi_TCCR0B = TCCR0B;
413  mozzi_OCR0A = OCR0A;
414  mozzi_TIMSK0 = TIMSK0;
415 }
