57 /*
58 ATmega328 technical manual, Section 12.7.4:
59 The dual-slope operation [of phase correct pwm] has lower maximum operation
60 frequency than single slope operation. However, due to the symmetric feature
61 of the dual-slope PWM modes, these modes are preferred for motor control
62 applications.
63 Due to the single-slope operation, the operating frequency of the
64 fast PWM mode can be twice as high as the phase correct PWM mode that use
65 dual-slope operation. This high frequency makes the fast PWM mode well suited
66 for power regulation, rectification, and DAC applications. High frequency allows
67 physically small sized external components (coils, capacitors)..
68 
69 DAC, that's us! Fast PWM.
70 
71 PWM frequency tests
72 62500Hz, single 8 or dual 16 bits, bad aliasing
73 125000Hz dual 14 bits, sweet
74 250000Hz dual 12 bits, gritty, if you're gonna have 2 pins, have 14 bits
75 500000Hz dual 10 bits, grittier
76 16384Hz single nearly 9 bits (original mode) not bad for a single pin, but
77 carrier freq noise can be an issue

279 static void tcConfigure(uint32_t sampleRate) {
280  // Enable GCLK for TCC2 and TC5 (timer counter input clock)
281  GCLK->CLKCTRL.reg = (uint16_t)(GCLK_CLKCTRL_CLKEN | GCLK_CLKCTRL_GEN_GCLK0 |
282  GCLK_CLKCTRL_ID(GCM_TC4_TC5));
283  while (GCLK->STATUS.bit.SYNCBUSY)
284  ;
285 
286  tcReset();
287 
288  // Set Timer counter Mode to 16 bits
289  TC5->COUNT16.CTRLA.reg |= TC_CTRLA_MODE_COUNT16;
290 
291  // Set TC5 mode as match frequency
292  TC5->COUNT16.CTRLA.reg |= TC_CTRLA_WAVEGEN_MFRQ;
293 
294  TC5->COUNT16.CTRLA.reg |= TC_CTRLA_PRESCALER_DIV1 | TC_CTRLA_ENABLE;
295 
296  TC5->COUNT16.CC[0].reg = (uint16_t)(SystemCoreClock / sampleRate - 1);
297  while (tcIsSyncing())
298  ;
299 
300  // Configure interrupt request
301  NVIC_DisableIRQ(TC5_IRQn);
302  NVIC_ClearPendingIRQ(TC5_IRQn);
303  NVIC_SetPriority(TC5_IRQn, 0);
304  NVIC_EnableIRQ(TC5_IRQn);
305 
306  // Enable the TC5 interrupt request
307  TC5->COUNT16.INTENSET.bit.MC0 = 1;
308  while (tcIsSyncing())
309  ;
310 }

314 // lookup table for fast pdm coding on 8 output bits at a time
315 static byte fast_pdm_table[]{0, 0b00010000, 0b01000100,
316  0b10010010, 0b10101010, 0b10110101,
317  0b11011101, 0b11110111, 0b11111111};

319 inline void writePDMCoded(uint16_t sample) {
320  static uint32_t lastwritten = 0;
321  static uint32_t nexttarget = 0;
322 
323  // We can write 32 bits at a time to the output buffer, typically, we'll do
324  // this either once of twice per sample
325  for (uint8_t words = 0; words < PDM_RESOLUTION; ++words) {
326  uint32_t outbits = 0;
327  for (uint8_t i = 0; i < 4; ++i) {
328  nexttarget += sample - lastwritten;
329  lastwritten =
330  nexttarget &
331  0b11110000000000000; // code the highest 3-and-a-little bits next.
332  // Note that sample only has 16 bits, while the
333  // highest bit we consider for writing is bit 17.
334  // Thus, if the highest bit is set, the next
335  // three bits cannot be.
336 #if (ESP_AUDIO_OUT_MODE == PDM_VIA_SERIAL)
337  U1F = fast_pdm_table[lastwritten >>
338  13]; // optimized version of: Serial1.write(...);
339 #else
340  outbits = outbits << 8;
341  outbits |= fast_pdm_table[lastwritten >> 13];
342 #endif
343  }
344 #if (ESP_AUDIO_OUT_MODE == PDM_VIA_I2S)
345  i2s_write_sample(outbits);
346 #endif
347  }
348 }

533 #define MAX_CARRIER_FREQ (F_CPU / (1 << AUDIO_BITS_PER_CHANNEL))
534 #if MAX_CARRIER_FREQ < AUDIO_RATE
535 #error Configured audio resolution is definitely too high at the configured audio rate (and the given CPU speed)
536 #elif MAX_CARRIER_FREQ < (AUDIO_RATE * 3)
537 #warning Configured audio resolution may be higher than optimal at the configured audio rate (and the given CPU speed)
538 #endif
539 
540 #if MAX_CARRIER_FREQ < (AUDIO_RATE * 5)
541  // Generate as fast a carrier as possible
542  audio_pwm_timer.setPrescaleFactor(1);
543 #else
544  // No point in generating arbitrarily high carrier frequencies. In fact, if
545  // there _is_ any headroom, give the PWM pin more time to swing from HIGH to
546  // LOW and BACK, cleanly
547  audio_pwm_timer.setPrescaleFactor((int)MAX_CARRIER_FREQ / (AUDIO_RATE * 5));
548 #endif
549  audio_pwm_timer.setOverflow(
550  1 << AUDIO_BITS_PER_CHANNEL); // Allocate enough room to write all
551  // intended bits
