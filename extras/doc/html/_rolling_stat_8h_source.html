18 // adapted from RunningStat, http://www.johndcook.com/standard_deviation.html
19 /** @ingroup sensortools
20 Calculates an approximation of the variance and standard deviation for a window of recent inputs.
21 @tparam T the type of numbers to use. Choose unsigned int, int , uint8_t, int8_t, or float
22 @tparam WINDOW_LENGTH how many recent input values to include in the calculations.
23 */
24 template <class T, int WINDOW_LENGTH>
25 class RollingStat
26 {
27 public:
28 
29  /** Constructor */
30  RollingStat() : _previous_mean(0), _mean(0), _variance(0), WINDOW_LENGTH_AS_RSHIFT((uint8_t)trailingZeros((unsigned long)WINDOW_LENGTH))
31  {}
32 
33 
34  /** Update the mean and variance given a new input value.
35  @param x the next input value
36  @note timing for unsigned int 10us, int 22us
37  */
38  void update(T x) {
39  _mean = rollingMean.next(x);
40  _variance = (((long)x - _previous_mean)*((long)x - _mean))>>WINDOW_LENGTH_AS_RSHIFT; // should really be div by (WINDOW_LENGTH-1), but exact values are not important for this application
41  _previous_mean = _mean;
42  }
43 
44 
45  /** Update the mean and variance given a new input value.
46  @param x the next input value
47  */
48  void update(int8_t x) {
49  _mean = rollingMean.next(x);
50  _variance = (((int)x - _previous_mean)*((int)x - _mean))>>WINDOW_LENGTH_AS_RSHIFT; // should really be div by (WINDOW_LENGTH-1), but exact values are not important for this application
51  _previous_mean = _mean;
52  }
53 
54 
55  /** Return the mean of the last WINDOW_LENGTH number of inputs.
56  @return mean
57  */
58  T getMean() const {
59  return _mean;
60  }
61 
62 
63  /** Return the approximate variance of the last WINDOW_LENGTH number of inputs.
64  @return variance
65  @note This should really be calculated using WINDOW_LENGTH-1, but sacrificing accuracy for speed
66  we use the power of two value of WINDOW_LENGTH.
67  */
68  T getVariance() const {
69  return _variance;
70  }
71 
72  /** Return the approximate standard deviation of the last WINDOW_LENGTH number of inputs.
73  @return standard deviation.
74  */
75  T getStandardDeviation() const {
76  return isqrt16(_variance);
77  }
78 
79 
80 
81 private:
82  T _previous_mean, _mean, _variance;
83  RollingAverage <T, WINDOW_LENGTH> rollingMean;
84  const uint8_t WINDOW_LENGTH_AS_RSHIFT;
85 };
