31 /** Sample is like Oscil, it plays a wavetable. However, Sample can be
32 set to play once through only, with variable start and end points,
33 or can loop, also with variable start and end points.
34 It defaults to playing once through the whole sound table, from start to finish.
35 @tparam NUM_TABLE_CELLS This is defined in the table ".h" file the Sample will be
36 using. The sound table can be arbitrary length for Sample.
37 It's important that NUM_TABLE_CELLS is either a literal number (eg. "8192") or a
38 defined macro, rather than a const or int, for the Sample to run fast enough.
39 @tparam UPDATE_RATE This will be AUDIO_RATE if the Sample is updated in
40 updateAudio(), or CONTROL_RATE if it's updated each time updateControl() is
41 called. It could also be a fraction of CONTROL_RATE if you are doing some kind
42 of cyclic updating in updateControl(), for example, to spread out the processor load.
43 @section int8_t2mozzi
44 Converting soundfiles for Mozzi.
45 There is a python script called int8_t2mozzi.py in the Mozzi/python folder.
46 The script converts raw sound data saved from a program like Audacity.
47 Instructions are in the int8_t2mozzi.py file.
48 */

159  /**
160  Returns the sample at the current phase position, or 0 if looping is off
161  and the phase overshoots the end of the sample. Updates the phase
162  according to the current frequency.
163  @return the next sample value from the table, or 0 if it's finished playing.
164  @todo in next(), incrementPhase() happens in a different position than for Oscil - check if it can be standardised
165  */
166  inline
167  int8_t next() { // 4us
168 
169  if (phase_fractional>endpos_fractional){
170  if (looping) {
171  phase_fractional = startpos_fractional + (phase_fractional - endpos_fractional);
172  }else{
173  return 0;
174  }
175  }
176  int8_t out;
177  if(INTERP==INTERP_LINEAR){
178  // WARNNG this is hard coded for when SAMPLE_F_BITS is 16
179  unsigned int index = phase_fractional >> SAMPLE_F_BITS;
180  out = FLASH_OR_RAM_READ<const int8_t>(table + index);
181  int8_t difference = FLASH_OR_RAM_READ<const int8_t>((table + 1) + index) - out;
182  int8_t diff_fraction = (int8_t)(((((unsigned int) phase_fractional)>>8)*difference)>>8); // (unsigned int) phase_fractional keeps low word, then>> for only 8 bit precision
183  out += diff_fraction;
184  }else{
185  out = FLASH_OR_RAM_READ<const int8_t>(table + (phase_fractional >> SAMPLE_F_BITS));
186  }
187  incrementPhase();
188  return out;
189  }

219  /** Set the oscillator frequency with an unsigned int.
220  This is faster than using a float, so it's useful when processor time is tight,
221  but it can be tricky with low and high frequencies, depending on the size of the
222  wavetable being used. If you're not getting the results you expect, try
223  explicitly using a float, or try setFreq_Q24n8.
224  @param frequency to play the wave table.
225  */
226  inline
227  void setFreq (int frequency) {
228  phase_increment_fractional = ((((unsigned long)NUM_TABLE_CELLS<<ADJUST_FOR_NUM_TABLE_CELLS)*frequency)/UPDATE_RATE) << (SAMPLE_F_BITS - ADJUST_FOR_NUM_TABLE_CELLS);
229  }
230 
231 
232  /** Set the sample frequency with a float. Using a float is the most reliable
233  way to set frequencies, -Might- be slower than using an int but you need either
234  this or setFreq_Q24n8 for fractional frequencies.
235  @param frequency to play the wave table.
236  */
237  inline
238  void setFreq(float frequency)
239  { // 1 us - using float doesn't seem to incur measurable overhead with the oscilloscope
240  phase_increment_fractional = (unsigned long)((((float)NUM_TABLE_CELLS * frequency)/UPDATE_RATE) * SAMPLE_F_BITS_AS_MULTIPLIER);
241  }

244  /** Set the frequency using Q24n8 fixed-point number format.
245  This might be faster than the float version for setting low frequencies
246  such as 1.5 Hz, or other values which may not work well with your table
247  size. Note: use with caution because it's prone to overflow with higher
248  frequencies and larger table sizes. An Q24n8 representation of 1.5 is 384
249  (ie. 1.5 * 256).
250  @param frequency in Q24n8 fixed-point number format.
251  */
252  inline
253  void setFreq_Q24n8(Q24n8 frequency)
254  {
255  //phase_increment_fractional = (frequency* (NUM_TABLE_CELLS>>3)/(UPDATE_RATE>>6)) << (F_BITS-(8-3+6));
256  phase_increment_fractional = (((((unsigned long)NUM_TABLE_CELLS<<ADJUST_FOR_NUM_TABLE_CELLS)>>3)*frequency)/(UPDATE_RATE>>6))
257  << (SAMPLE_F_BITS - ADJUST_FOR_NUM_TABLE_CELLS - (8-3+6));
258  }

272  /** phaseIncFromFreq() and setPhaseInc() are for saving processor time when sliding between frequencies.
273  Instead of recalculating the phase increment for each
274  frequency in between, you can just calculate the phase increment for each end
275  frequency with phaseIncFromFreq(), then use a Line to interpolate on the fly and
276  use setPhaseInc() to set the phase increment at each step. (Note: I should
277  really profile this with the oscilloscope to see if it's worth the extra
278  confusion!)
279  @param frequency for which you want to calculate a phase increment value.
280  @return the phase increment value which will produce a given frequency.
281  */
282  inline
283  unsigned long phaseIncFromFreq(unsigned int frequency)
284  {
285  return (((unsigned long)frequency * NUM_TABLE_CELLS)/UPDATE_RATE) << SAMPLE_F_BITS;
286  }
